Example

#include <stdio.h>  /* printf */
#include "mptrdef.h"  /* iptr, vptr, MPTRDEF_CONVERT */

int main()
{
    int a = 5;
    iptr ip = &a;
    vptr vp = MPTRDEF_CONVERT(void) ip;
    * MPTRDEF_CONVERT(int) vp = 7;

    printf("%d", *ip);

    return 0;
}

Output: 7
